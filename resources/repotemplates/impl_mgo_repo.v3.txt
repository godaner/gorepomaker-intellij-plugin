	if err != nil {
		return nil, err
	}
	return entity, nil
}

//查询${entityName}记录
func (this *${entityMongoRepoName}) Query${entityName}Many(query interface{}, options ...DB.DBOperationOption) (entities []*SELFENTITY.${entityName},err error) {
	opts := &DB.DBOperationOptions{}
	for _, o := range options {
		o(opts)
	}
	entities = []*SELFENTITY.${entityName}{}
	ops := []*mgooptions.FindOptions{}
	if opts.Select != nil {
		ops = append(ops, mgooptions.Find().SetProjection(opts.Select))
	}
	if opts.Sorts != nil {
		ops = append(ops, mgooptions.Find().SetSort(opts.Sorts))
	}
    if opts.Sortss != nil {
        ss := bsonx.Doc{}
        for _, s := range opts.Sortss {
            n := int32(1)
            if len(s)>=1{
                switch s[0] {
                case '+':
                    s = s[1:]
                case '-':
                    n = int32(-1)
                    s = s[1:]
                }
            }
			if s==""{
				this.Logger.Errorf("Query${entityName}Many : get sort field is nil !")
			}
            ss = append(ss, bsonx.Elem{Key:s, Value:bsonx.Int32(n)})
        }
        ops = append(ops, mgooptions.Find().SetSort(ss))
    }
	if opts.Limit >= 1 {
		ops = append(ops, mgooptions.Find().SetLimit(opts.Limit))
	}
	cursor, err := this.MongoClient.GetClient().Database(this.MongoDBName).Collection(SELFENTITY.${entityName}Col).Find(TC.TimeoutCtx.GetMongoOperate(0), query, ops...)
	if err != nil {
		return nil, err
	}
	if err := cursor.Err(); err != nil {
		return nil, err
	}
	defer func(){
	    err=cursor.Close(TC.TimeoutCtx.GetMongoOperate(0))
	    if err!=nil{
	        this.Logger.Errorf("Query${entityName}Many : cursor.Close err , err is : %v !",err.Error())
	    }
	}()
	for cursor.Next(TC.TimeoutCtx.GetMongoOperate(0)) {
		entity := SELFENTITY.${entityName}{}
		if err:=cursor.Decode(&entity);err!=nil{
		    this.Logger.Errorf("Query${entityName}Many : cursor.Decode err , err is : %v !",err.Error())
		}
		entities = append(entities, &entity)
	}
	return entities, nil

}
